## tips
(print)env                   環境変数等の確認


vi ~/.zshrc
man zshbuiltins             zshのbuilt-in cmdのhelp?
unalias run-help            run-helpはmanのalias設定(default)
autoload run-help           
HELPDIR=/usr/share/zsh/5.9/help 
ls /usr/share/zsh/5.9/help  zsh --version
alias help=run-help
source ~/.zshrc


ls *.txt                    .txtで終わるfileを参照
ls -lh                      file形式を記述
ls ..                       親ディレクトリの確認
ls /bin                     cmdの参照

which cmd                   cmdのディレクトリの参照


ctrl + k                    カーソル位置から行末まで一気にカット
ctrl + d                    カーソル位置から行末までカット
ctrl + w                    単語ごとにカット
ctrl + u                    カーソルの行をカット
ctrl + b                    1文字を戻る
ctrl + f                    1文字を進む
ctrl + -	                入力を戻す
ctrl + a	                行頭へ移動
ctrl + e	                行末へ移動
cmd + k                     画面をクリア


export PATH=$PATH:\
    $PWD/.package           iterm2が終了するまでPATHとしてenvに保存される
echo $PATH                  PATHの確認
echo $PATH | sed -e 's/:/\n/g'
PATH=$PATH:~/bin            通したいPATH名
echo "export PATH=$PATH:\
/Applications/通したいPATH\
/bin" >> ~/.zshrc
source ~/.zshrc             PATHの変更を反映
export                      変数を現在のシェル環境へ(環境変数が他のscriptから利用可能)
which cmd                   cmdのPATHを表示



/bin                        基本的なコマンド
/sbin                       管理者用コマンド
/usr/bin                    アプリケーション用の基本的なコマンド
/usr/sbin                   アプリケーション用の管理者用コマンド
/usr/local/bin              追加したアプリ用のコマンド



cat .file                   .fileを確認
open ~/.zshrc               .fileを開く(defaultのテキストエディタ)
touch ~/.zshrc              .fileを作成?
sysctl -n hw.ncpu           PCのCPUコア数の確認 
sysctl -n hw.logicalcpu_max
